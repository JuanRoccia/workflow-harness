_BANNER_WIDTH = 76  # caracteres entre los ║ bordes
# ─────────────────────────────────────────────────────────────────────────────


def _row(content: str = "", color: str = "cyan") -> str:
    """Arma una fila del banner con bordes ║ y padding exacto."""
    inner = content.ljust(_BANNER_WIDTH)
    return f"[{color}]║[/{color}]{inner}[{color}]║[/{color}]"


def print_banner(
    version: str = "2.0",
    target: str = None,
    output_file: str = None,
    modules: list = None,
    interface: str = None,
):
    W = _BANNER_WIDTH
    top = "╔" + "═" * W + "╗"
    bot = "╚" + "═" * W + "╝"
    sep = "║  " + "─" * (W - 4) + "  ║"
    empty = _row()

    console.print(f"[cyan]{top}[/cyan]")
    console.print(empty)

    # ASCII art — string literal fijo
    for line in _ASCII_LINES:
        console.print(f"[cyan]║[/cyan][bright_green]{line}[/bright_green][cyan]║[/cyan]")

    console.print(empty)

    # Línea título
    title = f"  [yellow]OpenHunter[/yellow] [white]— Modular Network Recon Framework[/white]  [dim]│  [/dim][magenta]v{version}[/magenta][dim]  │  by JuanRoccia[/dim]"
    console.print(f"[cyan]║[/cyan]{title}")

    # Subtítulo
    sub = "  [dim]nmap + recon-ng + AI-powered analysis[/dim]"
    console.print(f"[cyan]║[/cyan]{sub}")

    console.print(empty)
    console.print(f"[cyan]{sep}[/cyan]")

    # Módulos (dinámico)
    if modules:
        mods = "  ".join(f"● {m}" for m in modules)
        mod_line = f"  [white]Módulos:[/white]  [bright_green]{mods}[/bright_green]"
        console.print(f"[cyan]║[/cyan]{mod_line}")

    # Info de sesión (opcionales)
    if target:
        console.print(f"[cyan]║[/cyan]  [white]Target   :[/white]  [cyan]{target}[/cyan]")
    if interface:
        console.print(f"[cyan]║[/cyan]  [white]Interface:[/white]  [cyan]{interface}[/cyan]")
    if output_file:
        console.print(f"[cyan]║[/cyan]  [white]Output   :[/white]  [dim]{output_file}[/dim]")

    console.print(f"[cyan]{bot}[/cyan]")
    console.print()
